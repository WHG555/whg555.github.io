# samba配置文件 #
位置
>/etc/samba/smb.conf

# 建立共享 #
```
[show]
   comment = share folder
   browseable = yes
   path = /home/www/samba
   create mask = 0777
   directory mask = 0777
   valid users = whg
   force user = root
   force group = root
   public = yes
   available = yes
   writable = yes
```
